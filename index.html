  <script>

    // =========================================================

    // RAW-VALUE MODEL PARAMETERS

    // Replace all placeholder values with coefficients obtained from Stata.

    //

    // Stata model:

    // logit fenotipo_preesc c.wm_total c.tbv c.edadges_n c.edadges_c i.dd_sexo if sample_calc == 1

    //

    // Calculator formula:

    // logit(p adverse phenotype) = b0

    //   + b_wm  * total cerebral white matter volume

    //   + b_tbv * total brain volume

    //   + b_ga  * gestational age at birth

    //   + b_pma * postmenstrual age at MRI

    //   + b_sex * sex

    // =========================================================

    const MODEL = {

      b0: -3.954895,

      b_wm: -0.02471285,

      b_tbv: 0.00211581,

      b_ga: -0.06298582,

      b_pma: 0.22113674,

      b_sex: -1.1417114

    };


    function invlogit(x) {

      return 1 / (1 + Math.exp(-x));

    }


    function fmt(x, digits = 2) {

      if (!Number.isFinite(x)) return "—";

      return x.toFixed(digits);

    }


    function riskLabel(p) {

      if (p < 0.20) return "Lower estimated probability within this cohort";

      if (p < 0.40) return "Intermediate estimated probability within this cohort";

      return "Higher estimated probability within this cohort";

    }


    document.getElementById("riskForm").addEventListener("submit", function(event) {

      event.preventDefault();


      const gaBirth = Number(document.getElementById("gaBirth").value);

      const pmaMri = Number(document.getElementById("pmaMri").value);

      const tbv = Number(document.getElementById("tbv").value);

      const wm = Number(document.getElementById("wm").value);

      const sex = Number(document.getElementById("sex").value);


      const logit = MODEL.b0

        + MODEL.b_wm * wm

        + MODEL.b_tbv * tbv

        + MODEL.b_ga * gaBirth

        + MODEL.b_pma * pmaMri

        + MODEL.b_sex * sex;


      const probability = invlogit(logit);


      document.getElementById("probText").textContent = (probability * 100).toFixed(1) + "%";

      document.getElementById("riskText").textContent = riskLabel(probability);

      document.getElementById("logitValue").textContent = fmt(logit, 3);

      document.getElementById("wmValue").textContent = fmt(wm, 1);

      document.getElementById("tbvValue").textContent = fmt(tbv, 1);

    });

  </script>
